110 /**
111  * @brief Check the nucleation state depending on the nucleation strategy (see
112  * KKS_nucleation_strategy_)
113  *
114  * @tparam T
115  * @param indices_ph_1 List of nodes
116  * @param tp_gf_ph_1 The temperature at nodes defined by indices_ph_1
117  * @return std::set<int>
118  */
119 template <typename T>
120 std::set<int> KKS<T>::check_nucleation(CalphadBase<T>& CALPHAD, const std::set<int>& indices_ph_1,
121  const T& tp_gf_ph_1) {
122  std::set<int> indices_nucleation;
123 
124  switch (KKS_nucleation_strategy::from(this->KKS_nucleation_strategy_)) {
125  //
126  case KKS_nucleation_strategy::liquid_fraction: {
127  for (const auto& node : indices_ph_1) {
128  // Check only if equilibrium is found
129  if (CALPHAD.error_equilibrium_[node] == CalphadDefaultConstant::error_max) continue;
130  // Check if secondary phase is found
131  if (CALPHAD.elem_mole_fraction_by_phase_.contains(std::make_tuple(
132  node, this->KKS_secondary_phase_, this->element_removed_from_ic_))) {
133  indices_nucleation.insert(node);
134  }
135  }
136  break;
137  }
138  //
139  case KKS_nucleation_strategy::given_melting_temperature: {
140  for (const auto& node : indices_ph_1) {
141  // Check if temperature at node is greater than a given limit
142  if (tp_gf_ph_1(node) > this->given_melting_temperature_) {
143  indices_nucleation.insert(node);
144  }
145  }
146  break;
147  }
148  default: {
149  throw std::runtime_error(
150  "KKS<T>::check_nucleation_at_node: error in the choice of method used to compute "
151  "the nucleation. Available choices are : liquid_fraction (Default), "
152  "given_melting_temperature");
153  }
154  }
155  return indices_nucleation;
156 }

158 /**
159  * @brief Compute Calphad contributions needed by KKS resolution
160  *
161  * @tparam T The type of the template parameter (mfem::Vector or std::vector)
162  * @param CALPHAD The CALPHAD object for thermodynamic calculation
163  * @param dt The time step for the simulation.
164  * @param tp_gf The thermodynamic conditions at the current time step.
165  * @param tp_gf_old The thermodynamic conditions at the previous time step.
166  * @param phasefields_gf The phase-field at the current and previous time-step
167  * @param chemicalsystem The targeted chemical system
168  * @param x_gf The mole fraction of compoenent at the current and previous time-step
169  */
170 template <typename T>
171 void KKS<T>::execute_linearization(
172  CalphadBase<T>& CALPHAD, const int dt, const double time_step, const std::vector<T>& tp_gf,
173  const std::vector<T>& tp_gf_old, const std::tuple<std::string, T, T>& phasefields_gf,
174  const std::vector<std::tuple<std::string, std::string>>& chemicalsystem,
175  const std::vector<std::tuple<std::string, std::string, T, T>>& x_gf,
176  const std::vector<std::tuple<std::string, T>>& coordinates) {
177  // Number of chemical elements
178  const int nb_elem = chemicalsystem.size();
179  // Creation initial list of nodes
180  const size_t nb_nodes = this->CU_->get_size(tp_gf[0]);
181 
182  ////////////////////////////////////////////////////////
183  // Interpolation: H(phi, t+dt)=Hphi , H(phi, t)=Hphi_old
184  ////////////////////////////////////////////////////////
185  T Hphi(nb_nodes);
186  T Hphi_old(nb_nodes);
187 
188  ////////////////////////////////////////////////////////
189  // List of nodes by phases and within interface
190  ////////////////////////////////////////////////////////
191  const auto& [phase, phi_gf, phi_gf_old] = phasefields_gf;
192 
193  std::set<int> indices_ph_1;
194  std::set<int> indices_ph_2;
195  std::set<int> indices_inter;
196  for (unsigned int i = 0; i < nb_nodes; ++i) {
197  const double phi = std::clamp(phi_gf[i], 0.0, 1.0);
198  const double phi_old = std::clamp(phi_gf_old(i), 0.0, 1.0);
199  if (phi > 1 - this->KKS_threshold_) {
200  indices_ph_1.insert(i);
201  } else if (phi < this->KKS_threshold_) {
202  SlothInfo::debug("Execute_linearization: secondary_phase node ", i);
203  indices_ph_2.insert(i);
204 
205  } else {
206  SlothInfo::debug("Execute_linearization: interfacial node ", i);
207  indices_inter.insert(i);
208  }
209 
210  // Interpolation function at node
211  Hphi(i) = this->interpolation_func_->compute(std::span<const double>({phi}));
212  Hphi_old(i) = this->interpolation_func_->compute(std::span<const double>({phi_old}));
213  }
214 
215  // Save the number of two-phase nodes where KKS problem must be solved
216  if (this->KKS_enable_save_specialized_) {
217  int local_size = indices_inter.size();
218  int global_size;
219  MPI_Allreduce(&local_size, &global_size, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
220 
221  CALPHAD.time_specialized_.emplace(IterationKey(dt, time_step, dt * time_step),
222  SpecializedValue("KKS nodes[-]", global_size));
223  }
224 
225  // Lambda for equilibrium calculations performed in the pure phase or in the interface with a
226  // temperature deviation
227  auto calculate_interface =
228  [&](const std::set<int>& indices, const std::vector<T>& delta_tp_gf, const double increment,
229  const int id_incr,
230  std::map<std::tuple<int, std::string, std::string>, double>& chemical_potential_interface,
231  std::vector<std::tuple<std::string, std::string, double>> given_phase,
232  const std::string& primary_phase) {
233  std::vector<T> delta_tp = delta_tp_gf;
234  if (id_incr > -1) {
235  delta_tp[id_incr] += increment;
236  }
237  CALPHAD.execute(dt, indices, delta_tp, chemicalsystem, given_phase);
238 
239  for (const auto& in : indices) {
240  if (CALPHAD.error_equilibrium_[in] == CalphadDefaultConstant::error_max) continue;
241  for (const auto& elem : chemicalsystem) {
242  const auto& [elem1, unit] = elem;
243  const double mu = CALPHAD.chemical_potentials_.at(std::make_tuple(in, elem1));
244  chemical_potential_interface.emplace(std::make_tuple(in, elem1, primary_phase), mu);
245  }
246  }
247  };
248  //
249  // Lambda for equilibrium calculations performed in the pure phase with a deviation in
250  // composition
251  auto calculate_interface_x =
252  [&](const std::vector<T>& delta_tp_gf, const double increment, const int id_incr,
253  int index_el,
254  std::map<std::tuple<int, int, std::string, std::string>, double>&
255  chemical_potential_interface,
256  std::vector<std::tuple<std::string, std::string, double>> given_phase,
257  const std::string& primary_phase) {
258  std::vector<T> delta_tp = delta_tp_gf;
259  if (id_incr > -1) {
260  delta_tp[id_incr] += increment;
261  }
262 
263  CALPHAD.execute(dt, indices_inter, delta_tp, chemicalsystem, given_phase);
264  for (const auto& in : indices_inter) {
265  if (CALPHAD.error_equilibrium_[in] == CalphadDefaultConstant::error_max) continue;
266  for (const auto& elem : chemicalsystem) {
267  const auto& [elem1, unit] = elem;
268  const double mu = CALPHAD.chemical_potentials_.at(std::make_tuple(in, elem1));
269  chemical_potential_interface.emplace(
270  std::make_tuple(index_el, in, elem1, primary_phase), mu);
271  }
272  }
273  };
274 
275  ////////////////////////////////////////////////////////
276  /// Point of expansion (Tbar,Xbar)
277  ////////////////////////////////////////////////////////
278 
279  // Primary Phase : index _1
280  std::vector<T> bar_tp_gf_ph_1(tp_gf.size());
281  std::vector<T> pure_bar_tp_gf_ph_1(tp_gf.size());
282  // KKS_secondary_phase : index _2
283  std::vector<T> bar_tp_gf_ph_2(tp_gf.size());
284  std::vector<T> pure_bar_tp_gf_ph_2(tp_gf.size());
285  // Tbar
286  if (!this->KKS_temperature_scheme_) {
287  bar_tp_gf_ph_1[0] = tp_gf[0];
288  bar_tp_gf_ph_2[0] = tp_gf[0];
289  pure_bar_tp_gf_ph_1[0] = tp_gf[0];
290  pure_bar_tp_gf_ph_2[0] = tp_gf[0];
291  } else {
292  bar_tp_gf_ph_1[0] = tp_gf_old[0];
293  bar_tp_gf_ph_2[0] = tp_gf_old[0];
294  pure_bar_tp_gf_ph_1[0] = tp_gf_old[0];
295  pure_bar_tp_gf_ph_2[0] = tp_gf_old[0];
296  }
297  // Pressure
298  bar_tp_gf_ph_1[1] = tp_gf[1];
299  bar_tp_gf_ph_2[1] = tp_gf[1];
300  pure_bar_tp_gf_ph_1[1] = tp_gf[1];
301  pure_bar_tp_gf_ph_2[1] = tp_gf[1];
302 
303  // Composition : Xbar
304  MFEM_VERIFY(x_gf.size() > 0, "Error while getting molar fraction by phase");
305 
306  for (const auto& [elem, phase_elem, x_elem, x_elem_old] : x_gf) {
307  auto it = std::ranges::find_if(chemicalsystem,
308  [&elem](const auto& t) { return std::get<0>(t) == elem; });
309 
310  std::optional<int> id;
311  if (it != chemicalsystem.end()) {
312  id = std::distance(chemicalsystem.begin(), it);
313  }
314  MFEM_VERIFY(id.has_value(), "Error while getting the element");
315  const int id_value = id.value();
316  const bool is_primary_phase = (phase_elem == phase);
317  const bool is_secondary_phase = (phase_elem == this->KKS_secondary_phase_);
318 
319  if (is_primary_phase) {
320  if (elem != this->element_removed_from_ic_) {
321  bar_tp_gf_ph_1[id_value + 2] = x_elem;
322  } else {
323  bar_tp_gf_ph_1[id_value + 2] = tp_gf[id_value + 2];
324  }
325  pure_bar_tp_gf_ph_1[id_value + 2] = tp_gf[id_value + 2];
326 
327  } else if (is_secondary_phase) {
328  if (elem != this->element_removed_from_ic_) {
329  bar_tp_gf_ph_2[id_value + 2] = x_elem;
330 
331  // When scecondary phase is previously absent, initialize with primary phase
332  if (!indices_inter.empty()) {
333  for (const int id_new : indices_inter) {
334  if (bar_tp_gf_ph_2[id_value + 2](id_new) < this->xmin_tol_) {
335  bar_tp_gf_ph_2[id_value + 2](id_new) = tp_gf[id_value + 2](id_new);
336  }
337  }
338  }
339  } else {
340  bar_tp_gf_ph_2[id_value + 2] = tp_gf[id_value + 2];
341  }
342  pure_bar_tp_gf_ph_2[id_value + 2] = tp_gf[id_value + 2];
343 
344  } else {
345  const std::string& error_msg =
346  "Error while setting molar fraction at point approximation. Unknown phase " + phase_elem;
347  mfem::mfem_error(error_msg.c_str());
348  }
349  }
350 
351  ////////////////////////////////////////////////////////
352  /// Calculation of all thermodynamic contributions
353  ////////////////////////////////////////////////////////
354 
355  /////////////////////////
356  /// Primary phase
357  /////////////////////////
358  std::vector<std::tuple<std::string, std::string, double>> st_phase_12 = {
359  {phase, "entered", 0.}, {this->KKS_secondary_phase_, "entered", 0.}};
360 
361  if ((this->KKS_nucleation_started_ && this->KKS_freeze_nucleation_) ||
362  (pure_bar_tp_gf_ph_1[0][0] < this->KKS_temperature_threshold_) ||
363  (KKS_nucleation_strategy::from(this->KKS_nucleation_strategy_) !=
364  KKS_nucleation_strategy::liquid_fraction)) {
365  st_phase_12 = {{phase, "entered", 0.}};
366  }
367 
368  // In practice, two entered phase seems to be more stable than considering dormant phase
369  // {phase, "entered"}, {this->KKS_secondary_phase_, "dormant"}};
370  std::vector<std::tuple<std::string, std::string, double>> st_phase_1 = {{phase, "entered", 0.}};
371  std::vector<std::tuple<std::string, std::string, double>> st_phase_2 = {
372  {this->KKS_secondary_phase_, "entered", 0.}};
373 
374  calculate_interface(indices_ph_1, pure_bar_tp_gf_ph_1, 0., -1,
375  this->chemical_potentials_by_phase_, st_phase_12, phase);
376 
377  // Cancel before checking nucleation state
378  for (unsigned int i = 0; i < nb_nodes; ++i) {
379  // Must be zero except for nodes detected as nucleus in solid
380  CALPHAD.nucleus_[std::make_tuple(i, this->KKS_secondary_phase_)] = 0.;
381  // Must be zero except for nodes in interface.
382  CALPHAD.driving_forces_[std::make_tuple(i, this->KKS_secondary_phase_)] = 0.;
383  }
384 
385  if (!this->KKS_nucleation_started_) {
386  bool local_nucleation = false;
387  std::set<int> indices_nucleation =
388  this->check_nucleation(CALPHAD, indices_ph_1, pure_bar_tp_gf_ph_1[0]);
389 
390  // Nucleation detected ?
391  if (indices_nucleation.size() > 0) {
392  local_nucleation = true;
393  // If nucleation is not frozen once detected, local_nucleation must be false
394  local_nucleation &= this->KKS_freeze_nucleation_;
395  }
396 
397  // Create circular nucleus around the node where secondary phase is detected
398  // No need to check error_equilibrium because of already done when indices_nucleation is created
399  for (const auto& inuc : indices_nucleation) {
400  SlothInfo::debug("Nucleation detected at node ", inuc, " T ", pure_bar_tp_gf_ph_1[0](inuc));
401  // const double seed =
402  // CALPHAD.mole_fraction_of_phase_[std::make_tuple(inuc, this->KKS_secondary_phase_)];
403  const double seed = this->KKS_seed_;
404  const double nucleus_value = -seed / (time_step * this->KKS_mobility_for_seed_);
405  for (const auto& j : indices_ph_1) {
406  double rr = 0;
407 
408  for (const auto& [_, coord] : coordinates) {
409  const double coord_diff = coord[inuc] - coord[j];
410  rr += coord_diff * coord_diff;
411  }
412  if (std::sqrt(rr) < this->KKS_seed_radius_) {
413  SlothInfo::debug("Nucleation extended around node ", inuc, " with node ", j);
414  // double seed =
415  // CALPHAD.mole_fraction_of_phase_[std::make_tuple(node, this->KKS_secondary_phase_)];
416  // To enhance the chance to initiate the phase change
417  CALPHAD.nucleus_[std::make_tuple(j, this->KKS_secondary_phase_)] = nucleus_value;
418  }
419  }
420  }
421 
422  // Synchronization of nucleation_started across all MPI ranks
423  int local_flag = (local_nucleation ? 1 : 0);
424  int global_flag = 0;
425  MPI_Allreduce(&local_flag, &global_flag, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
426  this->KKS_nucleation_started_ = (global_flag == 1);
427  }
428 
429  /////////////////////////
430  /// Secondary phase (if exists)
431  /////////////////////////
432  if (!indices_ph_2.empty()) {
433  calculate_interface(indices_ph_2, pure_bar_tp_gf_ph_2, 0., -1,
434  this->chemical_potentials_by_phase_, st_phase_2,
435  this->KKS_secondary_phase_);
436  }
437 
438  /////////////////////////
439  /// Interface (if exists)
440  /////////////////////////
441  // 6 + 4 * (nelem-1) times more equilibrium calculation
442  //
443  if (!indices_inter.empty()) {
444  // T : equilibrium calculations in both phases (Tbar, Xbar)
445  calculate_interface(indices_inter, bar_tp_gf_ph_1, 0, -1, this->chemical_potentials_by_phase_,
446  st_phase_1, phase);
447  calculate_interface(indices_inter, bar_tp_gf_ph_2, 0., -1, this->chemical_potentials_by_phase_,
448  st_phase_2, this->KKS_secondary_phase_);
449 
450  // T-dT : equilibrium calculations in both phases (Tbar - deltaT, Xbar)
451  calculate_interface(indices_inter, bar_tp_gf_ph_1, -this->KKS_temperature_increment_, 0,
452  this->chemical_potentials_left_T_, st_phase_1, phase);
453 
454  calculate_interface(indices_inter, bar_tp_gf_ph_2, -this->KKS_temperature_increment_, 0,
455  this->chemical_potentials_left_T_, st_phase_2, this->KKS_secondary_phase_);
456 
457  // T+dT : equilibrium calculations in both phases (Tbar + deltaT, Xbar)
458  calculate_interface(indices_inter, bar_tp_gf_ph_1, this->KKS_temperature_increment_, 0,
459  this->chemical_potentials_right_T_, st_phase_1, phase);
460 
461  calculate_interface(indices_inter, bar_tp_gf_ph_2, this->KKS_temperature_increment_, 0,
462  this->chemical_potentials_right_T_, st_phase_2, this->KKS_secondary_phase_);
463 
464  // Loop over chemical system (except the reference element)
465  int ielem = 0;
466  for (const auto& [elem, unit] : chemicalsystem) {
467  if (elem != this->element_removed_from_ic_) {
468  // x + dx
469  // Equilibrium calculations in both phases at (Tbar, Xbar + delta X)
470  calculate_interface_x(bar_tp_gf_ph_1, this->KKS_composition_increment_, ielem + 2, ielem,
471  this->chemical_potentials_right_x_, st_phase_1, phase);
472 
473  calculate_interface_x(bar_tp_gf_ph_2, this->KKS_composition_increment_, ielem + 2, ielem,
474  this->chemical_potentials_right_x_, st_phase_2,
475  this->KKS_secondary_phase_);
476 
477  // x - dx
478  // Equilibrium calculations in both phases at (Tbar, Xbar - delta X)
479  calculate_interface_x(bar_tp_gf_ph_1, -this->KKS_composition_increment_, ielem + 2, ielem,
480  this->chemical_potentials_left_x_, st_phase_1, phase);
481 
482  calculate_interface_x(bar_tp_gf_ph_2, -this->KKS_composition_increment_, ielem + 2, ielem,
483  this->chemical_potentials_left_x_, st_phase_2,
484  this->KKS_secondary_phase_);
485  }
486  ielem++;
487  }
488 
489  ////////////////////////////////////////////////////////
490  /// Solve linear systems
491  ////////////////////////////////////////////////////////
492  // Allocate
493  mfem::Array<int> offsets(3);
494  offsets[0] = 0;
495  offsets[1] = nb_elem - 1;
496  offsets[2] = 2 * nb_elem - 2;
497 
498  mfem::BlockVector bb(offsets);
499  mfem::BlockVector deltaX_phase(offsets);
500  mfem::Vector HphiNode(nb_elem - 1);
501  mfem::Vector one_minus_HphiNode(nb_elem - 1);
502 
503  std::unique_ptr<mfem::BlockMatrix> AA = std::make_unique<mfem::BlockMatrix>(offsets, offsets);
504 
505  auto solver = std::make_shared<mfem::GMRESSolver>();
506  solver->SetOperator(*AA);
507  solver->SetAbsTol(this->kks_abs_tol_solver_);
508  solver->SetRelTol(this->kks_rel_tol_solver_);
509  solver->SetMaxIter(this->kks_max_iter_solver_);
510  solver->SetPrintLevel(this->kks_print_level_solver_);
511 
512  for (const auto& node : indices_inter) {
513  // Solve KKS only if an equilibrium is found. Otherwise, solution at previous time-step will
514  // be taken into account
515  if (CALPHAD.error_equilibrium_[node] == CalphadDefaultConstant::error_max) continue;
516 
517  //

568  " with x ", current_x, " and x_old ", old_x);
569  SlothInfo::debug("Calculation of xdH at node ", node, " = ", xdH, " for elem ", elem,
570  " with xs_old ", bar_tp_gf_ph_1[jelem + 2](node), " xl_old ",
571  bar_tp_gf_ph_2[jelem + 2](node), " Hphi(node) ", current_Hphi,
572  " and Hphi_old(node) ", old_Hphi);

576  ielem++;
577  }
578  jelem++;
579  }

595  d1 = 0.;

599  *Al *= -1.;

605

607  auto A11 = std::make_unique<mfem::SparseMatrix>(one_minus_HphiNode);
608

610  AA->SetBlock(1, 1, A11.get());

615  // Solve system
616  //
617  // mfem::BlockVector deltaX_phase(offsets);
618  // Check result of linear system
619  if (Verbosity::Debug <= verbosityLevel) {
620  SlothInfo::print("BEfore solving, KKS linear system at node (A X = B):", node);
621  AA->Print();
622  deltaX_phase.Print();
623  bb.Print();
624  }

627  // Check result of linear system
628  if (Verbosity::Debug <= verbosityLevel) {
629  SlothInfo::print("KKS linear system at node (A X = B):", node);
630  AA->Print();
631  deltaX_phase.Print();
632  bb.Print();
633  }

636  *Al *= -1.;
637

639  /// Recover thermodynamic contribution
640  ////////////////////////////////////////////////////////
641  const mfem::Vector delta_XS = deltaX_phase.GetBlock(0);
642  const mfem::Vector delta_XL = deltaX_phase.GetBlock(1);
643  ////////////////////////////////////////
644  // Composition by phase
645  ////////////////////////////////////////
646  int i = 0;

648  double sumS = 0.;
649  double sumL = 0.;

651  if (elem != this->element_removed_from_ic_) {

655  node, this->KKS_secondary_phase_, elem)] = bar_tp_gf_ph_2[i + 2](node) + delta_XL(ie);

657  sumL += CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(
658  node, this->KKS_secondary_phase_, elem)];
659  ie++;
660  }

662  }
663  // Last component in each phase
664  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(
665  node, phase, this->element_removed_from_ic_)] = 1. - sumS;
666  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(

694  DeltaXlAlDeltaXl *= 0.5;

696  auto DeltaXlMl = ml * delta_XL;
697  gs += DeltaXsMs;
698  gs += DeltaXsAsDeltaXs;
699  gl += DeltaXlMl;
700  gl += DeltaXlAlDeltaXl;

703  CALPHAD.energies_of_phases_[std::make_tuple(node, phase, "GM")] = gs;
704  CALPHAD.energies_of_phases_[std::make_tuple(node, this->KKS_secondary_phase_, "GM")] = gl;
705

707  CALPHAD.driving_forces_[std::make_tuple(node, phase)] =
708  -CALPHAD.energies_of_phases_[std::make_tuple(node, phase, "GM")];
709  CALPHAD.driving_forces_[std::make_tuple(node, this->KKS_secondary_phase_)] =
710  -CALPHAD.energies_of_phases_[std::make_tuple(node, this->KKS_secondary_phase_, "GM")];
711 
712  i = 0;
713  ie = 0;
714  for (const auto& [elem, unit] : chemicalsystem) {
715  if (elem != this->element_removed_from_ic_) {
716  SlothInfo::debug("Check at node ", node, " for elem ", elem);
717  SlothInfo::debug("deltaX to obtain ", deltaX(ie), " deltaX predicted ",

724  node, this->KKS_secondary_phase_, elem)] *

726

728  " ; secondary phase", dmu_l(ie));
729

732

750  " ; secondary phase ",

756 /**
757  * @brief Second order finite difference approximation
758  *
759  * @tparam T The type of the template parameter (mfem::Vector or std::vector)
760  * @param chemicalsystem The targeted chemical system
761  * @param phase The phase for which derivatives are computed
762  * @param node The node where calculation are done
763  * @return mfem::SparseMatrix*
764  */
765 template <typename T>
766 std::unique_ptr<mfem::SparseMatrix> KKS<T>::get_A4linearKKS(
767  const std::vector<std::tuple<std::string, std::string>>& chemicalsystem,
768  const std::string& phase, const int node) {
769  const int nb_elem = chemicalsystem.size();
770  auto AA = std::make_unique<mfem::SparseMatrix>(nb_elem - 1, nb_elem - 1);
771 
772  // Diagonal
773  // d2dxdx = d(mu_x - mu-n)/dx =[ (mu_x - mu-n)(x+dx) - (mu_x - mu-n)(x-dx) ] / 2dx
774  // elem_id corresponds to index of variable with delta_x, index of derivative
775  int elem_id = 0;
776  int vid = 0;
777  for (const auto& [elem, unit] : chemicalsystem) {
778  if (elem != this->element_removed_from_ic_) {
779  double d2gd2x =
780  this->chemical_potentials_right_x_[std::make_tuple(elem_id, node, elem, phase)] -
781  this->chemical_potentials_right_x_[std::make_tuple(
782  elem_id, node, this->element_removed_from_ic_, phase)] -
783  (this->chemical_potentials_left_x_[std::make_tuple(elem_id, node, elem, phase)] -
784  this->chemical_potentials_left_x_[std::make_tuple(
785  elem_id, node, this->element_removed_from_ic_, phase)]);
786  d2gd2x /= 2. * this->KKS_composition_increment_;
787  AA->Set(vid, vid, d2gd2x);
788  vid++;
789  }
790  elem_id++;
791  }
792  // Off-diagonal (and symmetry)
793  elem_id = 0;
794  vid = 0;
795  for (const auto& [ielem, unit] : chemicalsystem) {
796  if (ielem != this->element_removed_from_ic_) {
797  int elem_jd = 0;
798  int vjd = 0;
799  ///////////
800  // row i
801  for (const auto& [jelem, unit] : chemicalsystem) {
802  if (jelem != this->element_removed_from_ic_) {
803  ///////////
804  // col j
805  //
806  // d2dxdy = 0.5 d(mu_x - mu-n)/dy + 0.5 d(mu_y - mu-n)/dx
807  if (ielem != jelem) {
808  double dmuxdy =
809  (this->chemical_potentials_right_x_[std::make_tuple(elem_jd, node, ielem, phase)] -
810  this->chemical_potentials_right_x_[std::make_tuple(
811  elem_jd, node, this->element_removed_from_ic_, phase)]) -
812  ((this->chemical_potentials_left_x_[std::make_tuple(elem_jd, node, ielem, phase)] -
813  this->chemical_potentials_left_x_[std::make_tuple(
814  elem_jd, node, this->element_removed_from_ic_, phase)]));
815  dmuxdy /= 2. * this->KKS_composition_increment_;
816  double dmuydx =
817  (this->chemical_potentials_right_x_[std::make_tuple(elem_id, node, jelem, phase)] -
818  this->chemical_potentials_right_x_[std::make_tuple(
819  elem_id, node, this->element_removed_from_ic_, phase)]) -
820  ((this->chemical_potentials_left_x_[std::make_tuple(elem_id, node, jelem, phase)] -
821  this->chemical_potentials_left_x_[std::make_tuple(
822  elem_id, node, this->element_removed_from_ic_, phase)]));
823  dmuydx /= 2. * this->KKS_composition_increment_;
824 
825  double d2gdxdy = 0.5 * dmuxdy + 0.5 * dmuydx;
826 
827  AA->Set(vid, vjd, d2gdxdy);
828  }
829  vjd++;
830  ///////////
831  }
832  elem_jd++;
833  }
834  ///////////
835  vid++;
836  }
837  elem_id++;
838  }
839 
840  return AA;
841 }
