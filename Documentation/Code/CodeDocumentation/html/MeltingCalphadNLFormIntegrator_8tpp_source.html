96 /**
97  * @brief Check the presence of driving forces variables for two phases and stores them in a
98  * container 'dgm_'
99  *
100  * @tparam VARS Template parameter defining the variables used
101  * in the integrator.
102  */
103 template <class VARS>
104 void MeltingCalphadNLFormIntegrator<VARS>::check_driving_forces() {
105  bool primary_phase_found = false;
106  bool secondary_phase_found = false;
107 
108  for (std::size_t i = 0; i < this->aux_infos_.size(); ++i) {
109  const auto& variable_info = this->aux_infos_[i];
110  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
111 
112  size_t vsize = variable_info.size();
113  MFEM_VERIFY(vsize > 0,
114  "MeltingCalphadNLFormIntegrator<VARS, SCHEME, ENERGY, "
115  "MOBI,INTERPOLATION>: at least "
116  "one additionnal information is expected for auxiliary variables associated with "
117  "this integrator");
118  const std::string& symbol = toLowerCase(variable_info.back());
119 
120  if (calphad_outputs::from(symbol) != calphad_outputs::dgm) continue;
121  MFEM_VERIFY(vsize == 2,
122  "Error while getting driving forces. Two additional informations are excepted "
123  ": the name of the phase and the symbol 'dgm'");
124 
125  if (variable_info[0] == this->primary_phase_) {
126  this->dgm_.insert(this->dgm_.begin(), this->aux_gf_[i]);
127  primary_phase_found = true;
128  } else if (variable_info[0] == this->secondary_phase_) {
129  this->dgm_.emplace_back(this->aux_gf_[i]);
130  secondary_phase_found = true;
131  }
132  }
133  MFEM_VERIFY(primary_phase_found && secondary_phase_found,
134  "Both primary and secondary driving forces must be set.");
135 }

173 /**
174  * @brief Compute the value of the phase change at integration point
175  *
176  * @remark Written for two phase
177  *
178  * @tparam VARS Template parameter defining the variables used in the integrator.
179  *
180  * @param Tr Element transformation.
181  * @param ir Integration point
182  * @param blk Index of the block.
183  * @param u value of the current solution at ip
184  * @param un value of the previous solution at ip
185  *
186  * @return The computed phase change at integration point
187  */
188 template <class VARS>
189 double MeltingCalphadNLFormIntegrator<VARS>::get_phase_change_at_ip(
190  mfem::ElementTransformation& Tr, const mfem::IntegrationPoint& ir,
191  [[maybe_unused]] unsigned int blk, [[maybe_unused]] const double u,
192  [[maybe_unused]] const double un) {
193  double primary_dgm = -1.;
194  double secondary_dgm = -1.;
195  if (this->dgm_.size() == 2) {
196  primary_dgm = this->dgm_[0].GetValue(Tr, ir);
197  secondary_dgm = this->dgm_[1].GetValue(Tr, ir);
198  }
199  constexpr double epsilon = 1.e-12;
200  return (std::abs(primary_dgm) > epsilon && std::abs(secondary_dgm) > epsilon)
201  ? (this->alpha_ * (secondary_dgm - primary_dgm))
202  : 0.;
203 }
