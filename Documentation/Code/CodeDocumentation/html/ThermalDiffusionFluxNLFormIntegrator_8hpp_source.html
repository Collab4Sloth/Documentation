114 void ThermalDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency() {
115  std::vector<mfem::ParGridFunction> aux_gf = this->get_aux_gf();
116  std::vector<std::vector<std::string>> aux_infos = this->get_aux_infos();
117 
118  //==============================================================
119  // Get Temperature and thermal diffusivity (aux. variables)
120  //==============================================================
121  bool temperature_found = false;
122  bool diffusivity_found = false;
123  for (std::size_t i = 0; i < aux_infos.size(); ++i) {
124  const auto& variable_info = aux_infos[i];
125 
126  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
127 
128  size_t vsize = variable_info.size();
129 
130  MFEM_VERIFY(vsize > 1,
131  "ThermalDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: at least "
132  "two additionnal information are expected for auxiliary variables associated with "
133  "this integrator");
134  const std::string& symbol = toUpperCase(variable_info.back());
135 
136  if (toUpperCase(symbol) == "T") {
137  this->temp_gf_ = aux_gf[i];
138  temperature_found = true;
139  } else if (toLowerCase(symbol) == "lambda") {
140  // Diffusivity can be directly supplied within this integrator or overloaded by considering a
141  // child class.
142  this->diffu_gf_ = aux_gf[i];
143  this->diffusivity_found_ = true;
144  }
145 
146  if (temperature_found && diffusivity_found) break;
147  }
148 
149  MFEM_VERIFY(temperature_found,
150  "ThermalDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
151  "Temperature not found. Please check your data.");
152 }
