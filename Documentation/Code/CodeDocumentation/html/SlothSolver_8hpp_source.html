186  using TT = std::decay_t<decltype(solv)>;
187  using PP = std::decay_t<decltype(prec)>;
188  UtilsSolvers::print_solver_name(solv);
189  UtilsSolvers::print_prec_name(prec);
190  if constexpr (!std::is_same_v<PP, sptr<std::monostate>>) {
191  if constexpr (is_in_variant_v<TT, VIterativeSolver>) {
192  MFEM_VERIFY((is_in_variant_v<PP, VIterativePrecond>),
193  "SetPrecondSolver: IterativeSolver objects can only be associated with an "
194  "IterativePreconditionner objects");
195  if constexpr (is_in_variant_v<PP, VIterativePrecond>) {
196  SlothInfo::debug("SetPrecondSolver: setting iterative preconditionner");
197  UtilsSolvers::setter_mfem(solv, prec);
198  }
199  }
200  if constexpr (is_in_variant_v<TT, VHypreSolver>) {
201  MFEM_VERIFY((is_in_variant_v<PP, VHyprePrecond>),
202  "SetPrecondSolver: HypreSolver objects can only be associated with an "
203  "HyprePreconditionner objects");
204  if constexpr (is_in_variant_v<PP, VHyprePrecond>) {
205  SlothInfo::debug("SetPrecondSolver: setting hypre preconditionner (not hypre smoother)");
206  UtilsSolvers::setter_hypre(solv, prec);

211  if constexpr (!std::is_same_v<TT, sptr<std::monostate>>) {
