208  template <typename Solv, typename Prec>
209  inline void operator()(Solv&& solv, Prec&& prec) {
210  using TT = std::decay_t<decltype(solv)>;
211  using PP = std::decay_t<decltype(prec)>;
212  UtilsSolvers::print_solver_name(solv);
213  UtilsSolvers::print_prec_name(prec);
214  if constexpr (!std::is_same_v<PP, sptr<std::monostate>>) {
215  if constexpr (is_in_variant_v<TT, VIterativeSolver>) {
216  MFEM_VERIFY((is_in_variant_v<PP, VIterativePrecond>),
217  "SetPrecondSolver: IterativeSolver objects can only be associated with an "
218  "IterativePreconditionner objects");
219  if constexpr (is_in_variant_v<PP, VIterativePrecond>) {
220  SlothInfo::debug("SetPrecondSolver: setting iterative preconditionner");
221  UtilsSolvers::setter_mfem(solv, prec);
222  }
223  }
224  if constexpr (is_in_variant_v<TT, VHypreSolver>) {
225  MFEM_VERIFY((is_in_variant_v<PP, VHyprePrecond>),
226  "SetPrecondSolver: HypreSolver objects can only be associated with an "
227  "HyprePreconditionner objects");
228  if constexpr (is_in_variant_v<PP, VHyprePrecond>) {
229  SlothInfo::debug("SetPrecondSolver: setting hypre preconditionner (not hypre smoother)");
230  UtilsSolvers::setter_hypre(solv, prec);
231  }
232  }
233  }
234 
235  if constexpr (!std::is_same_v<TT, sptr<std::monostate>>) {
236  SlothInfo::debug("SetPrecondSolver: setting operator ");
237  solv->SetOperator(this->op);
238  }
239  }
