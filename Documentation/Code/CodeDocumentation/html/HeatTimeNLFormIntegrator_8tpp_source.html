68 /**

75  *
76  * @remark By default, A and B are equal to one. This method specializes the coefficients for the

79  * @tparam VARS Template parameter defining the variables used in the integrator.
80  *
81  */
82 template <class VARS>
83 void HeatTimeNLFormIntegrator<VARS>::get_coefficients() {
84  for (unsigned int i = 0; i < this->nb_blk_; i++) {
85  if (this->get_coefficient(i, GlossaryType::Concentration, 0).has_value()) {
86  this->coefficient_A.add(*(this->get_coefficient(i, GlossaryType::Concentration, 0)));
87  }
88  if (this->get_coefficient(i, GlossaryType::HeatCapacity, 0).has_value()) {
89  this->coefficient_B.add(*(this->get_coefficient(i, GlossaryType::HeatCapacity, 0)));
