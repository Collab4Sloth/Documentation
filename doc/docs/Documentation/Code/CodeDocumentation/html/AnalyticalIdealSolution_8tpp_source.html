100 /**
101  * @brief Compute the CALPHAD contributions
102  *
103  * @tparam T
104  * @param tp_gf
105  * @param chemical_system
106  * @param output_system
107  */
108 template <typename T>
109 void AnalyticalIdealSolution<T>::compute(
110  const std::set<int>& list_nodes, const std::vector<T>& tp_gf,
111  const std::vector<std::tuple<std::string, std::string>>& chemical_system) {
112  // Let us assume an ideal mixing solution
113  const std::string& phase = "SOLUTION";
114  const std::vector<std::string> energy_names = {"G"};
115  std::vector<double> tp_gf_at_node(tp_gf.size());
116 
117  const auto temperature_sort_method =
118  this->params_.template get_param_value_or_default<std::string>("temperature_sort_method",
119  "No");
120  const auto pressure_sort_method =
121  this->params_.template get_param_value_or_default<std::string>("pressure_sort_method", "No");
122 
123  std::vector<int> sorted_n_t_p =
124  this->CU_->sort_nodes(tp_gf[0], tp_gf[1], temperature_sort_method, pressure_sort_method);
125 
126  // Remove elements from list_nodes in sorted_n_t_p
127  // Usefull if the targeted list is lower than the complete list (eg. KKS problems)
128  sorted_n_t_p.erase(
129  std::remove_if(sorted_n_t_p.begin(), sorted_n_t_p.end(),
130  [&list_nodes](int node) { return list_nodes.find(node) == list_nodes.end(); }),
131  sorted_n_t_p.end());
132 
133  // Process CALPHAD calculations for each node
134  for (const auto& id : sorted_n_t_p) {
135  // Populate tp_gf_at_node for the current node
136  std::transform(tp_gf.begin(), tp_gf.end(), tp_gf_at_node.begin(),
137  [&id](const T& vec) { return vec[id]; });
138  const auto Temp = tp_gf_at_node[0];
139  const auto& epsilon = 1.e-10;
140  const auto x = std::max(epsilon, std::min(1. - epsilon, tp_gf_at_node[2]));
141  const auto& elem = std::get<0>(chemical_system[0]);
142 
143  // Energy
144  for (const auto& energy_name : energy_names) {
145  const auto& key = std::make_tuple(id, phase, energy_name);
146  this->energies_of_phases_[key] =
147  Physical::R * Temp * (x * std::log(x) + (1. - x) * std::log(1. - x));
148  }
149 
150  // Chemical potential
151  this->chemical_potentials_[std::make_tuple(id, elem)] =
152  Physical::R * Temp * (std::log(x) - std::log(1. - x));
153 
154  // Diffusion chemical potential
155  this->diffusion_chemical_potentials_[std::make_tuple(id, elem)] =
156  Physical::R * Temp * (std::log(x) - std::log(1. - x));
157 
158  // Molar fraction
159  const auto& key = std::make_tuple(id, phase, elem);
160  this->elem_mole_fraction_by_phase_[key] = x;
161  }
162 }
163 /**
164  * @brief Check the consistency of outputs required for the current Calphad problem
165  *
166  * @tparam T
167  * @param output_system
168  */
169 template <typename T>
170 void AnalyticalIdealSolution<T>::check_variables_consistency(
171  std::vector<std::tuple<std::vector<std::string>, std::reference_wrapper<T>>>& output_system) {
172  for (auto& [output_infos, output_value] : output_system) {
173  const std::string& output_type = output_infos.back();
174 
175  // Fill output with the relevant values
176  switch (calphad_outputs::from(output_type)) {
177  case calphad_outputs::gm:
178  case calphad_outputs::h:
179  case calphad_outputs::hm: {
180  MFEM_VERIFY(false, "AnalyticalIdealSolution is only built for mu, x and g.\n");
181 
182  SlothInfo::debug("Output not available for this Calphad problem: ", output_type);
183  break;
184  }
185  // (Optional) handle allowed cases explicitly
186  case calphad_outputs::mu:
187  case calphad_outputs::x:
188  case calphad_outputs::g:
189  // process normally
190  break;
191 
192  default:
193  MFEM_VERIFY(false, "Unhandled output type.\n");
194  break;
195  }
196  }
