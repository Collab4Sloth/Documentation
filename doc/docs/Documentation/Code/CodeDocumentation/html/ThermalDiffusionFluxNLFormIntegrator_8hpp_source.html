114  std::vector<mfem::ParGridFunction> aux_gf = this->get_aux_gf();
115  std::vector<std::vector<std::string>> aux_infos = this->get_aux_infos();
116 
117  //==============================================================
118  // Get Temperature and thermal diffusivity (aux. variables)
119  //==============================================================
120  bool temperature_found = false;
121  bool diffusivity_found = false;
122  for (std::size_t i = 0; i < aux_infos.size(); ++i) {
123  const auto& variable_info = aux_infos[i];
124 
125  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
126 
127  size_t vsize = variable_info.size();
128 
129  MFEM_VERIFY(vsize > 1,
130  "ThermalDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: at least "
131  "two additionnal information are expected for auxiliary variables associated with "
132  "this integrator");
133  const std::string& symbol = toUpperCase(variable_info.back());
134 
135  if (toUpperCase(symbol) == "T") {
136  this->temp_gf_ = aux_gf[i];
137  temperature_found = true;
138  } else if (toLowerCase(symbol) == "lambda") {
139  // Diffusivity can be directly supplied within this integrator or overloaded by considering a
140  // child class.
141  this->diffu_gf_ = aux_gf[i];
142  this->diffusivity_found_ = true;
143  }
144 
145  if (temperature_found && diffusivity_found) break;
146  }
147 
148  MFEM_VERIFY(temperature_found,
149  "ThermalDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
150  "Temperature not found. Please check your data.");
151 }
