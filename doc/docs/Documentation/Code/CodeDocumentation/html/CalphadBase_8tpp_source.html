89 /**
90  * @brief High-level method for managing equilibrium calculations
91  *
92  * @tparam T
93  * @param dt The current iteration of the simulation
94  * @param time_step The current time-step of the simulation
95  * @param tp_gf The thermodynamic condition for equilibrium calculations
96  * @param chemicalsystem The targeted chemical system
97  * @param output_system The output variables at the current time-step
98  * @param previous_output_system The output variables at the previous time-step
99  * @param phase_field_gf The phase-field variable at the current and previous time-step
100  * @param tp_gf_old The thermodynamic condition for equilibrium calculations at the previous
101  * time-step
102  * @param x_gf The molar fractions of the elements at the current and previous time-step
103  * @param coordinates The coordinates of each node of the mesh (see nucleation)
104  */
105 template <typename T>
106 void CalphadBase<T>::global_execute(
107  const int dt, const double time_step, const std::vector<T>& tp_gf,
108  const std::vector<std::tuple<std::string, std::string>>& chemicalsystem,
109  std::vector<std::tuple<std::vector<std::string>, std::reference_wrapper<T>>>& output_system,
110  const std::vector<std::tuple<std::vector<std::string>, mfem::Vector>>& previous_output_system,
111  std::optional<const std::tuple<std::string, T, T>> phase_field_gf,
112  std::optional<const std::vector<T>> tp_gf_old,
113  std::optional<const std::vector<std::tuple<std::string, std::string, T, T>>> x_gf,
114  std::optional<const std::vector<std::tuple<std::string, T>>> coordinates) {
115  const size_t nb_nodes = this->CU_->get_size(tp_gf[0]);
116  // Reinitialize containers
117  this->clear_containers();
118 
119  // Execute
120  if (!this->is_KKS_) {
121  // Creation list of nodes
122  std::set<int> list_nodes;
123  for (unsigned int i = 0; i < nb_nodes; ++i) {
124  list_nodes.insert(i);
125  }
126  this->execute(dt, list_nodes, tp_gf, chemicalsystem);
127  } else {
128  // Verify that all required fields are available for KKS execution
129  MFEM_VERIFY(phase_field_gf.has_value(),
130  "Error: phase_fields_gf is required for KKS execution.");
131  MFEM_VERIFY(tp_gf_old.has_value(), "Error: tp_gf_old is required for KKS execution.");
132  MFEM_VERIFY(x_gf.has_value(), "Error: x_gf is required for KKS execution.");
133  MFEM_VERIFY(coordinates.has_value(), "Error: coordinates is required for KKS execution.");
134 
135  // Execute KKS linearization
136  this->KKS_->execute_linearization(*this, dt, time_step, tp_gf, *tp_gf_old, *phase_field_gf,
137  chemicalsystem, *x_gf, *coordinates);
138  }
139  // Use specific CALPHAD C++ containers to update output_system
140  this->update_outputs(dt, nb_nodes, output_system, previous_output_system);
141 }

167 /**
168  * @brief Update the outputs on the basis of results stored in containers
169  *
170  * @tparam T
171  * @param dt The current iteration of the simulation
172  * @param nb_nodes The number of nodes (local mesh)
173  * @param output_system The output variables at the current time-step
174  * @param previous_output_system The output variables at the previous time-step
175  *
176  * @remark A default value is used when the output is not calculated (but required). The value at
177  * the previous time-step can be used in case of convergence error (for GEM).
178  *
179  */
180 template <typename T>
181 void CalphadBase<T>::update_outputs(
182  [[maybe_unused]] const int dt, const size_t nb_nodes,
183  std::vector<std::tuple<std::vector<std::string>, std::reference_wrapper<T>>>& output_system,
184  const std::vector<std::tuple<std::vector<std::string>, mfem::Vector>>& previous_output_system) {
185  Catch_Time_Section("CalphadBase<T>::update_outputs");
186 
187  ////////////////////
188  // Update outputs //
189  ////////////////////
190  T output(nb_nodes);
191  auto get_or_default = [&](const auto& map, const auto& key, auto default_value) {
192  if (map.contains(key)) {
193  return map.at(key);
194  } else {
195  return default_value;
196  }
197  };
198  int id_output = -1;
199  double default_value = 0.;
200 
201  for (auto& [output_infos, output_value] : output_system) {
202  id_output++;
203  const std::string& output_type = output_infos.back();
204 
205  // Fill output with the relevant values
206  switch (calphad_outputs::from(output_type)) {
207  // Chemical potential of the elements
208  case calphad_outputs::mu: {
209  const std::string& output_elem = output_infos[1];
210  for (unsigned int i = 0; i < nb_nodes; ++i) {
211  default_value = 0.;
212  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
213  default_value = std::get<1>(previous_output_system[id_output])(i);
214  }
215  output[i] = get_or_default(this->chemical_potentials_, std::make_tuple(i, output_elem),
216  default_value);
217  }
218  break;
219  }
220  // Diffusion chemical potentials (reference is the element removed from initial condition)
221  case calphad_outputs::dmu: {
222  const std::string& output_elem = output_infos[1];
223  for (unsigned int i = 0; i < nb_nodes; ++i) {
224  default_value = 0.;
225  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
226  default_value = std::get<1>(previous_output_system[id_output])(i);
227  }
228  output[i] = get_or_default(this->diffusion_chemical_potentials_,
229  std::make_tuple(i, output_elem), default_value);
230  }
231  break;
232  }
233  // Molar fraction of the phases
234  case calphad_outputs::xph: {
235  const std::string& output_phase = output_infos[1];
236  for (unsigned int i = 0; i < nb_nodes; ++i) {
237  default_value = 0.;
238  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
239  default_value = std::get<1>(previous_output_system[id_output])(i);
240  }
241  output[i] = get_or_default(this->mole_fraction_of_phase_,
242  std::make_tuple(i, output_phase), default_value);
243  }
244  break;
245  }
246  // Molar fraction of the element by phase
247  case calphad_outputs::xp: {
248  const std::string& output_elem = output_infos[1];
249  const std::string& output_phase = output_infos[2];
250  for (unsigned int i = 0; i < nb_nodes; ++i) {
251  default_value = 0.;
252  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
253  default_value = std::get<1>(previous_output_system[id_output])(i);
254  }
255  output[i] = get_or_default(this->elem_mole_fraction_by_phase_,
256  std::make_tuple(i, output_phase, output_elem), default_value);
257  }
258  break;
259  }
260  // Site fractions
261  case calphad_outputs::y: {
262  const std::string& output_cons = output_infos[1];
263  const int& output_sub = std::stoi(output_infos[2]);
264  const std::string& output_phase = output_infos[3];
265  for (unsigned int i = 0; i < nb_nodes; ++i) {
266  default_value = 0.;
267  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
268  default_value = std::get<1>(previous_output_system[id_output])(i);
269  }
270  output[i] = get_or_default(this->site_fraction_,
271  std::make_tuple(i, output_phase, output_cons, output_sub),
272  default_value);
273  }
274  break;
275  }
276  // Gibbs energy
277  case calphad_outputs::g: {
278  const std::string& output_phase = output_infos[1];
279  for (unsigned int i = 0; i < nb_nodes; ++i) {
280  default_value = 0.;
281  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
282  default_value = std::get<1>(previous_output_system[id_output])(i);
283  }
284  output[i] = get_or_default(this->energies_of_phases_,
285  std::make_tuple(i, output_phase, "G"), default_value);
286  }
287  break;
288  }
289  // Molar Gibbs energy
290  case calphad_outputs::gm: {
291  const std::string& output_phase = output_infos[1];
292  for (unsigned int i = 0; i < nb_nodes; ++i) {
293  default_value = 0.;
294  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
295  default_value = std::get<1>(previous_output_system[id_output])(i);
296  }
297  output[i] = get_or_default(this->energies_of_phases_,
298  std::make_tuple(i, output_phase, "GM"), default_value);
299  }
300  break;
301  }
302  // Enthalpy
303  case calphad_outputs::h: {
304  const std::string& output_phase = output_infos[1];
305  for (unsigned int i = 0; i < nb_nodes; ++i) {
306  default_value = 0.;
307  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
308  default_value = std::get<1>(previous_output_system[id_output])(i);
309  }
310  output[i] = get_or_default(this->energies_of_phases_,
311  std::make_tuple(i, output_phase, "H"), default_value);
312  }
313  break;
314  }
315  // Molar enthalpy
316  case calphad_outputs::hm: {
317  const std::string& output_phase = output_infos[1];
318  for (unsigned int i = 0; i < nb_nodes; ++i) {
319  default_value = 0.;
320  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
321  default_value = std::get<1>(previous_output_system[id_output])(i);
322  }
323  output[i] = get_or_default(this->energies_of_phases_,
324  std::make_tuple(i, output_phase, "HM"), default_value);
325  }
326  break;
327  }
328  // Driving force
329  case calphad_outputs::dgm: {
330  const std::string& output_phase = output_infos[1];
331  for (unsigned int i = 0; i < nb_nodes; ++i) {
332  default_value = 0.;
333  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
334  default_value = std::get<1>(previous_output_system[id_output])(i);
335  }
336  output[i] = get_or_default(this->driving_forces_, std::make_tuple(i, output_phase),
337  default_value);
338  }
339  break;
340  }
341  // Heat capacity (per mole?)
342  case calphad_outputs::cp: {
343  for (unsigned int i = 0; i < nb_nodes; ++i) {
344  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
345  output[i] = std::get<1>(previous_output_system[id_output])(i);
346  } else {
347  output[i] = this->heat_capacity_[i];
348  }
349  }
350  break;
351  }
352  // Mobility by element (if a model exists)
353  case calphad_outputs::mob: {
354  const std::string& output_phase = output_infos[1];
355  const std::string& output_elem = output_infos[2];
356  for (unsigned int i = 0; i < nb_nodes; ++i) {
357  default_value = 0.;
358  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
359  default_value = std::get<1>(previous_output_system[id_output])(i);
360  }
361  output[i] = get_or_default(this->mobilities_,
362  std::make_tuple(i, output_phase, output_elem), default_value);
363  }
364  break;
365  }
366  // Seed for starting nucleation in KKS studies
367  case calphad_outputs::nucleus: {
368  const std::string& output_phase = output_infos[1];
369  for (unsigned int i = 0; i < nb_nodes; ++i) {
370  default_value = 0.;
371  if (this->error_equilibrium_[i] == CalphadDefaultConstant::error_max) {
372  default_value = std::get<1>(previous_output_system[id_output])(i);
373  }
374  output[i] =
375  get_or_default(this->nucleus_, std::make_tuple(i, output_phase), default_value);
376  }
377  break;
378  }
379  // Convergence error (used with GEM)
380  case calphad_outputs::error: {
381  for (unsigned int i = 0; i < nb_nodes; ++i) {
382  output[i] = this->error_equilibrium_[i];
383  }
384  break;
385  }
386  case calphad_outputs::x: {
387  // Nothing to do
388  break;
389  }
390  }
391  // Update the referenced output vector
392  output_value.get() = output;
393  }
394 }
