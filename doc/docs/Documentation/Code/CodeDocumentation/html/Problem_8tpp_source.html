284 void Problem<OPE, VAR, PST>::save_specialized(bool must_be_saved) {
285  int rank = mfem::Mpi::WorldRank();
286  if (rank == 0) {
287  if (must_be_saved) {
288  if (!this->oper_.get_time_specialized().empty()) {
289  this->pst_.save_specialized(this->oper_.get_time_specialized());
290  }
291 
292  if (!this->oper_.get_time_iso_specialized().empty()) {
293  for (const auto& [var_name, variable_time_iso_specialized] :
294  this->oper_.get_time_iso_specialized()) {
295  std::string str = var_name + "_iso_computation.csv";
296  this->pst_.save_specialized(variable_time_iso_specialized, str);
297  }
298  }
299  }
300  }
301  if (must_be_saved) {
302  this->oper_.clear_time_specialized();
303  this->oper_.clear_iso_time_specialized();
304  }

306 /**
307  * @brief Saves variables according PostProcessing settings
308  *
309  * @tparam OPE
310  * Operator type defining the nonlinear PDE.
311  * @tparam VAR
312  * Primary variable container type.
313  * @tparam PST
314  * PostProcessing.
315  *
316  * @param iter Current iteration
317  * @param current_time Current time
318  *
319  */
320 template <class OPE, class VAR, class PST>
321 void Problem<OPE, VAR, PST>::post_processing(const int& iter, const double& current_time) {
322  const auto nvars = this->variables_.get_variables_number();
323  std::vector<mfem::Vector> u_vect;
324  const double current_time_step = this->oper_.get_current_time_step();
325 
326  // Isovalue to compute by variable
327  const std::map<std::string, double> map_iso_value = this->pst_.get_iso_val_to_compute();
328 
329  // Integral value to compute by variable
330  const std::map<std::string, std::tuple<double, double>> map_integral =
331  this->pst_.get_integral_to_compute();
332 
333  for (unsigned int iv = 0; iv < nvars; iv++) {
334  auto vv = this->variables_[iv];
335  auto solution = vv.get_analytical_solution();
336  auto unk = vv.get_unknown();
337  auto unk_name = vv.getVariableName();
338 
339  // Errors
340  if (solution != nullptr) {
341  auto solution_func = solution.get();
342  this->oper_.ComputeError(iter, current_time, current_time_step, iv, unk_name, unk,
343  *solution_func);
344  }
345 
346  // Isovalues
347  if (!map_iso_value.empty() && map_iso_value.contains(unk_name)) {
348  const double iso_value = map_iso_value.at(unk_name);
349  this->oper_.ComputeIsoVal(iter, current_time, current_time_step, iv, unk_name, unk,
350  iso_value);
351  }
352 
353  // Integral
354  if (!map_integral.empty() && map_integral.contains(unk_name)) {
355  const auto& [lower_bound, upper_bound] = map_integral.at(unk_name);
356  this->oper_.ComputeIntegral(iter, current_time, current_time_step, iv, unk_name, unk,
357  lower_bound, upper_bound);
358  }
359  u_vect.emplace_back(std::move(unk));
360  }
361  // Compute Energies is required (True by default)
362  if (this->pst_.get_enable_compute_energies()) {
363  this->oper_.ComputeEnergies(iter, current_time, current_time_step, u_vect);
364  }
365  // Save variables for visualization
366  ProblemBase<VAR, PST>::post_processing(iter, current_time);
367 
368  // Save specialized values at each time-step if required
369  bool must_be_saved = this->pst_.get_enable_save_specialized_at_iter();
370  this->save_specialized(must_be_saved);
371 }

373 /**
374  * @brief Do a time-step by calling Step method of the ODE
375  *
376  * @tparam OPE
377  * Operator type defining the nonlinear PDE.
378  * @tparam VAR
379  * Primary variable container type.
380  * @tparam PST
381  * PostProcessing.
382  *
383  * @param next_time Next time.
384  * @param current_time Current time.
385  * @param current_time_step Current time-step.
386  * @param iter Current iteration.
387  * @param vect_unk Vector of pointers towards the unknowns.
388  * @param unks_info Vector containing information associated with the unknowns.
389  */
390 template <class OPE, class VAR, class PST>
391 void Problem<OPE, VAR, PST>::do_time_step(
392  double& next_time, const double& current_time, double current_time_step, const int iter,
393  std::vector<std::unique_ptr<mfem::Vector>>& vect_unk,
394  [[maybe_unused]] const std::vector<std::vector<std::string>>& unks_info) {
395  const size_t unk_size = vect_unk.size();
396 
397  this->set_time_coefficients(next_time);
398 
399  this->oper_.setGeometry(this->geometry_);
400  this->oper_.solve(vect_unk, next_time, current_time, current_time_step, iter);
401 
402  // Store the solution into a temporary mfem::Vector that will be used during updating stage, if
403  // calculation converges
404  for (size_t i = 0; i < unk_size; i++) {
405  this->unknown_.emplace_back(*(vect_unk[i]));
406  }
407 }
