186 template <class VAR, class PST>
187 void ProblemBase<VAR, PST>::update() {
188  MFEM_VERIFY(this->variables_.get_variables_number() == this->unknown_.size(),
189  "Error while updating the variables: the number of variables must be equal to the "
190  "size of the unknown vector");
191  for (std::size_t i = 0; i < this->variables_.get_variables_number(); i++) {
192  auto& var = this->variables_[i];
193  var.update(this->unknown_[i]);
194  }
195  this->unknown_.clear();
196 }

198 /**
199  * @brief Execute step
200  *
201  * @tparam VAR Type representing the problem Variables.
202  * @tparam PST Type representing the post-processing.
203  * @param iter Current iteration number.
204  * @param next_time Next simulation time (Current time+ current time step).
205  * @param current_time Current simulation time.
206  * @param current_time_step Current time-step.
207  */
208 template <class VAR, class PST>
209 void ProblemBase<VAR, PST>::execute(const int& iter, double& next_time, const double& current_time,
210  const double& current_time_step) {
211  int rank = mfem::Mpi::WorldRank();
212  if (rank == 0) {
213  SlothInfo::verbose(" ============================== ");
214  SlothInfo::verbose(" ==== Problem : ", this->name_);
215  SlothInfo::verbose(" ============================== ");
216  }
217 
218  std::vector<std::unique_ptr<mfem::Vector>> vect_unk;
219  std::vector<std::vector<std::string>> vect_unk_info;
220 
221  size_t num_vars = this->variables_.getVariables().size();
222  vect_unk.reserve(num_vars);
223  vect_unk_info.reserve(num_vars);
224 
225  for (auto& var : this->variables_.getVariables()) {
226  vect_unk.push_back(std::make_unique<mfem::Vector>(var.get_unknown()));
227 
228  auto& unk_info = vect_unk_info.emplace_back(var.get_additional_variable_info());
229  unk_info.insert(unk_info.begin(), var.getVariableName());
230  }
231 
232  this->do_time_step(next_time, current_time, current_time_step, iter, vect_unk, vect_unk_info);
233 
234  if (this->convergence_ != nullptr && iter > 1) {
235  this->check_convergence(vect_unk);
236  }
237 }

284 /**
285  * @brief Check convergence of variables at the current time-step
286  *
287  * @tparam VAR Type representing the problem Variables.
288  * @tparam PST Type representing the post-processing.
289  * @param unks Unknown at the current time-step.
290  */
291 template <class VAR, class PST>
292 void ProblemBase<VAR, PST>::check_convergence(
293  const std::vector<std::unique_ptr<mfem::Vector>>& unks) {
294  this->var_convergence_.clear();
295 
296  std::vector<PhysicalConvergence> vect_convergence = this->convergence_->getPhysicalConvergence();
297 
298  int j = -1;
299  for (auto& var : this->variables_.getVariables()) {
300  j++;
301  const auto& prev_unk = var.get_last();
302  mfem::Vector& unk = *(unks[j]);
303  const auto& [local_converged, local_criterion] =
304  vect_convergence[j].getPhysicalConvergence(unk, prev_unk);
305 
306  int local_conv_int = static_cast<int>(local_converged);
307  int global_conv_int = 0;
308  double global_criterion = 0.0;
309 
310  MPI_Allreduce(&local_conv_int, &global_conv_int, 1, MPI_INT, MPI_LAND, MPI_COMM_WORLD);
311  MPI_Allreduce(&local_criterion, &global_criterion, 1, MPI_DOUBLE, MPI_MAX, MPI_COMM_WORLD);
312 
313  // int rank = mfem::Mpi::WorldRank();
314  // if (rank == 0) {
315  bool has_converged = static_cast<bool>(global_conv_int);
316  this->var_convergence_.emplace_back(
317  std::make_tuple(var.getVariableName(), has_converged, global_criterion));
318  // }
319  }
320 }
