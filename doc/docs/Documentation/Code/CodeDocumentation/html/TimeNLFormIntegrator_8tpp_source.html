72 /**

91  this->get_coefficients();

94  this->get_coefficients();

97  for (std::size_t i = 0; i < this->aux_infos_.size(); ++i) {

110  // Temperature scaling for mobility

112  const auto& variable_info = this->aux_infos_[i];
113  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
114  size_t vsize = variable_info.size();
115 
116  MFEM_VERIFY(vsize >= 1,
117  "TimeNLFormIntegrator<VARS>: at least "
118  "one additionnal information is expected for auxiliary variables associated with "
119  "this integrator");

121  if (symbol == "T") {

123  break;

128 /**
129  * @brief Retrieve and store the coefficients required by the TimeDerivative integrator.
130  *
131  * This method collects two default coefficients:
132  * - 'coefficient_A' stores the first coefficient,
133  * - 'coefficient_B' stores the second coefficient,
134  *
135  *
136  * @remark By default, A and B are equal to one. This method can be overridden in derived classes to
137  * provide custom behavior for retrieving coefficients (see HeatTimeDerivative). Coefficient objects
138  * of type GlossaryType::ExplicitTime can be also used for this class (see Glossary::ExplicitTime_A
139  * and Glossary::ExplicitTime_B used wth explicit solver)
140  *
141  * @tparam VARS Template parameter defining the variables used in the integrator.
142  *
143  */
144 template <class VARS>
145 void TimeNLFormIntegrator<VARS>::get_coefficients() {
146  for (unsigned int i = 0; i < this->nb_blk_; i++) {
147  auto coef_A = this->get_coefficient(i, GlossaryType::ExplicitTime, 0);
148 
149  if (!coef_A.has_value()) {
150  this->coefficient_A.add(Coefficient(Glossary::Default, 1.0));
151  } else {
152  if ((*coef_A).is_implicit()) {
153  MFEM_VERIFY(
154  false,
155  "Implicit coefficient for TimeDerivative integrator not implemented yet. Please "
156  "check your data.");
157  }
158  this->coefficient_A.add(*coef_A);
159  }
160  auto coef_B = this->get_coefficient(i, GlossaryType::ExplicitTime, 1);
161 
162  if (!coef_B.has_value()) {
163  this->coefficient_B.add(Coefficient(Glossary::Default, 1.0));
164  } else {
165  if ((*coef_B).is_implicit()) {
166  MFEM_VERIFY(
167  false,
168  "Implicit coefficient for TimeDerivative integrator not implemented yet. Please "
169  "check your data.");
170  }
171  this->coefficient_B.add(*coef_B);
172  }
173  }
174 }

176 /**
177  * @brief Assemble the element-level residual vector for the nonlinear problem.
178  *
179  * This method computes the residual vector by element
180  *
181  *
182  * @tparam VARS Template parameter defining the variables used in the integrator.
183  *
184  * @param el Array of pointers to finite elements.
185  * @param Tr Element transformation.
186  * @param elfun Array of local finite element solution vectors.
187  * @param elvect Array of vectors where the computed element residual contributions
188  * will be stored.
189  *
190  * @note Users typically do not call this function directly; it is invoked
191  * internally during the assembly of the global nonlinear form.
192  */
193 template <class VARS>
194 void TimeNLFormIntegrator<VARS>::AssembleElementVector(
195  const mfem::Array<const mfem::FiniteElement*>& el, mfem::ElementTransformation& Tr,
196  const mfem::Array<const mfem::Vector*>& elfun, const mfem::Array<mfem::Vector*>& elvect) {
197  int num_blocks = el.Size();
198  std::vector<double> u_values(2 * num_blocks);
199  std::vector<double> vaux_gf_at_ip(this->vaux_gf_.size());
200  for (int blk = 0; blk < num_blocks; ++blk) {
201  // Catch_Time_Section("TimeNLFormIntegrator:AssembleElementVector");
202  int nd = el[blk]->GetDof();
203  int dim = el[blk]->GetDim();
204  gradPsi.SetSize(nd, dim);
205  Psi.SetSize(nd);
206  gradU.SetSize(dim);
207  // elvect.SetSize(nd);
208  elvect[blk]->SetSize(nd);
209  *elvect[blk] = 0.;
210 
211  const mfem::IntegrationRule* ir =
212  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
213  // elvect = 0.0;
214  for (int i = 0; i < ir->GetNPoints(); i++) {
215  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
216  el[blk]->CalcShape(ip, Psi); //
217  Tr.SetIntPoint(&ip);
218 
219  const auto& u = *elfun[blk] * Psi;
220  // Get aux values at ip TODO(cci) (move in method)
221  for (size_t k = 0; k < vaux_gf_.size(); ++k) {
222  vaux_gf_at_ip[k] = vaux_gf_[k].GetValue(Tr, ip);
223  }
224  // Get values
225  for (int off_blk = 0; off_blk < num_blocks; ++off_blk) {
226  u_values[off_blk] = (*elfun[off_blk]) * Psi;
227  u_values[off_blk + num_blocks] = this->u_old_[off_blk].GetValue(Tr, ip);
228  }
229  double coef_a =
230  this->compute_coefficient(this->coefficient_A[blk], std::span<const double>(u_values),
231  std::span<const double>(vaux_gf_at_ip));
232  double coef_b =
233  this->compute_coefficient(this->coefficient_B[blk], std::span<const double>(u_values),
234  std::span<const double>(vaux_gf_at_ip));
235 
236  double weight_coef = ip.weight * Tr.Weight();
237  if (this->isAxisymmetric()) {
238  weight_coef *= AxiCylindricalCoefficient().Eval(Tr, ip);
239  }
240  const double ww = coef_a * coef_b * u * weight_coef;
241  add(*elvect[blk], ww, Psi, *elvect[blk]);
242  }
243  }
244 }

248  *
249  * This method computes the Jacobian matrix by element.
250  *
251  * @tparam VARS Template parameter defining the variables used in the integrator.
252  *
253  * @param el Array of pointers to finite elements.
254  * @param Tr Element transformation.
255  * @param elfun Array of local finite element solution vectors.
256  * @param elmat Array of dense matrices where the computed element Jacobian contributions
257  * will be stored.
258  *
259  * @note Users typically do not call this function directly; it is invoked
260  * internally during the assembly of the global nonlinear form.
261  */
262 template <class VARS>
263 void TimeNLFormIntegrator<VARS>::AssembleElementGrad(
264  const mfem::Array<const mfem::FiniteElement*>& el, mfem::ElementTransformation& Tr,
265  const mfem::Array<const mfem::Vector*>& elfun,
266  const mfem::Array2D<mfem::DenseMatrix*>& elmats) {
267  // Catch_Time_Section("TimeNLFormIntegrator::AssembleElementGrad");
268  int num_blocks = el.Size();
269  std::vector<double> u_values(2 * num_blocks);
270  std::vector<double> vaux_gf_at_ip(this->vaux_gf_.size());
271  for (int blk = 0; blk < num_blocks; ++blk) {
272  // int nd = el.GetDof();
273  // int dim = el.GetDim();
274  int nd = el[blk]->GetDof();
275  int dim = el[blk]->GetDim();
276 
277  gradPsi.SetSize(nd, dim);
278  Psi.SetSize(nd);
279  // elmat.SetSize(nd);
280  elmats(blk, blk)->SetSize(nd);
281  *elmats(blk, blk) = 0.0;
282 
283  const mfem::IntegrationRule* ir =
284  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
285 
286  // elmat = 0.0;
287  for (int i = 0; i < ir->GetNPoints(); i++) {
288  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
289  el[blk]->CalcShape(ip, Psi);
290  Tr.SetIntPoint(&ip);
291  // Get aux values at ip TODO(cci) (move in method)
292  for (size_t k = 0; k < vaux_gf_.size(); ++k) {
293  vaux_gf_at_ip[k] = vaux_gf_[k].GetValue(Tr, ip);
294  }
295  // Get values
296  for (int off_blk = 0; off_blk < num_blocks; ++off_blk) {
297  u_values[off_blk] = (*elfun[off_blk]) * Psi;
298  u_values[off_blk + num_blocks] = this->u_old_[off_blk].GetValue(Tr, ip);
299  }
300  double coef_a =
301  this->compute_coefficient(this->coefficient_A[blk], std::span<const double>(u_values),
302  std::span<const double>(vaux_gf_at_ip));
303  double coef_b =
304  this->compute_coefficient(this->coefficient_B[blk], std::span<const double>(u_values),
305  std::span<const double>(vaux_gf_at_ip));
306  double weight_coef = ip.weight * Tr.Weight();
307  if (this->isAxisymmetric()) {
308  weight_coef *= AxiCylindricalCoefficient().Eval(Tr, ip);
309  }
310  double fun_val = coef_a * coef_b * weight_coef;
311  AddMult_a_VVt(fun_val, Psi, *elmats(blk, blk));
312  }
313  }
314 
315  // Off-diagonal
316  for (int blk = 0; blk < num_blocks; ++blk) {
317  for (int off_blk = 0; off_blk < num_blocks; ++off_blk) {
318  if (off_blk != blk) {
319  int nd = el[blk]->GetDof();
320 
321  // elmat.SetSize(nd);
322  elmats(blk, off_blk)->SetSize(nd);
323  *elmats(blk, off_blk) = 0.0;
324  }
325  }
326  }
