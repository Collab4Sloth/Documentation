173 template <class... Args>
174 void ExprTkCoefficient::build_functional(Args&&... function_variable_names) {
175  this->math_expression_ = "";
176  // Constant mathematical expression
177  if constexpr (sizeof...(function_variable_names) == 1) {
178  this->math_expression_ = {std::forward<Args>(function_variable_names)...};
179  this->variable_names_.clear();
180  } else {
181  // Variable mathematical expression
182  this->variable_names_ = {std::forward<Args>(function_variable_names)...};
183 
184  this->math_expression_ = this->variable_names_[0];
185  if (!this->variable_names_.empty()) {
186  this->variable_names_.erase(this->variable_names_.begin());
187  }
188  }
189  // Variables
190  for (const auto& var : this->variable_names_) {
191  this->reference_map_[var] = 0.0; // initialize all to 0
192  this->symbol_table_.add_variable(var, this->reference_map_[var]);
193  }
194 
195  // Constant parameters
196  this->symbol_table_.add_constants();
197  this->expression_parser_.register_symbol_table(this->symbol_table_);
198 
199  if (!this->parser_.compile(this->math_expression_, this->expression_parser_)) {
200  this->ParserCoefficientError(this->math_expression_);
201  }
202 }
