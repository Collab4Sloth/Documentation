143 void InterDiffusionCoefficient::get_property(
144  std::vector<std::tuple<std::vector<std::string>, std::reference_wrapper<mfem::Vector>>>&
145  output_system,
146  std::vector<std::tuple<std::vector<std::string>, mfem::Vector>> input_system) {
147  primary_mob_gf_.clear();
148  secondary_mob_gf_.clear();
149 
150  x_gf_.clear();
151  phi_gf_.clear();
152  for (const auto& [aux_infos, aux_gf] : input_system) {
153  if (!aux_infos.empty()) {
154  const int size = aux_infos.size();
155  const std::string_view type_info_view(aux_infos.back());
156 
157  if (!type_info_view.empty() && type_info_view.starts_with("mob")) {
158  const std::string phase_info(aux_infos[size - 3]);
159  const std::string elem_info(aux_infos[size - 2]);
160  if (phase_info == this->primary_phase_) {
161  this->primary_mob_gf_.emplace(toUpperCase(elem_info), aux_gf);
162 
163  } else if (phase_info == this->secondary_phase_) {
164  this->secondary_mob_gf_.emplace(toUpperCase(elem_info), aux_gf);
165  }
166  }
167  if (!type_info_view.empty() && type_info_view == "x") {
168  const std::string elem_info(aux_infos[size - 2]);
169  this->x_gf_.emplace(toUpperCase(elem_info), aux_gf);
170  }
171  if (!type_info_view.empty() && type_info_view == "phi") {
172  const std::string elem_info(aux_infos[size - 2]);
173  this->phi_gf_.emplace_back(aux_gf);
174  }
175  }
176  }
177  const int size_gf = this->primary_mob_gf_.at(this->first_component_).Size();
178 
179  auto mult_vectors = [](const std::vector<double>& v1, const std::vector<double>& v2,
180  std::vector<double>& out) {
181  std::ranges::transform(v1, v2, out.begin(), [](double x, double y) { return x * y; });
182  };
183 
184  auto sum_vectors = [](const std::vector<double>& v1, const std::vector<double>& v2,
185  std::vector<double>& out) {
186  std::ranges::transform(v1, v2, out.begin(), [](double x, double y) { return x + y; });
187  };
188 
189  auto scale_vector = [](std::vector<double>& v, double scalar) {
190  std::ranges::transform(v, v.begin(), [=](double x) { return x * scalar; });
191  };
192 
193  auto add_scalar = [](std::vector<double>& v, double scalar) {
194  std::ranges::transform(v, v.begin(), [=](double x) { return x + scalar; });
195  };
196 
197  std::vector<std::vector<double>> vmob;
198  vmob.reserve(this->primary_mob_gf_.size());
199 
200  //
201  std::vector<double> phi(size_gf, 1.0);
202  if (!this->single_phase_) {
203  for (int k = 0; k < size_gf; k++) {
204  phi[k] = std::clamp(this->phi_gf_[0](k), 0.0, 1.0);
205  }
206  }
207  std::vector<double> xo(this->x_gf_.at(this->first_component_).Size(), 0.);
208  for (int k = 0; k < this->x_gf_.at(this->first_component_).Size(); k++) {
209  xo[k] = this->x_gf_.at(this->first_component_)(k);
210  }
211 
212  std::vector<double> mobo(this->primary_mob_gf_.at(this->first_component_).Size(), 0.);
213  if (this->single_phase_) {
214  for (int k = 0; k < this->primary_mob_gf_.at(this->first_component_).Size(); k++) {
215  mobo[k] = this->primary_mob_gf_.at(this->first_component_)(k);
216  }
217  } else {
218  for (int k = 0; k < this->primary_mob_gf_.at(this->first_component_).Size(); k++) {
219  mobo[k] = phi[k] * this->primary_mob_gf_.at(this->first_component_)(k);
220 
221  mobo[k] += (1. - phi[k]) * this->secondary_mob_gf_.at(this->first_component_)(k);
222  }
223  }
224 
225  std::vector<double> mob(size_gf, 0.0);
226  std::vector<double> sum_mob(size_gf, 0.0);
227 
228  for (const auto& [compo, mobi] : this->primary_mob_gf_) {
229  if (compo == this->first_component_) continue;
230 
231  std::vector<double> xj(size_gf, 0.0);
232  std::vector<double> mobj(size_gf, 0.0);
233 
234  if (compo == this->last_component_) {
235  add_scalar(xj, -1.0);
236  for (const auto& [other_compo, _] : this->primary_mob_gf_) {
237  if (other_compo != this->last_component_) {
238  for (int k = 0; k < this->x_gf_.at(other_compo).Size(); k++) {
239  xj[k] += this->x_gf_.at(other_compo)(k);
240  }
241  }
242  }
243  scale_vector(xj, -1.0);
244  } else {
245  for (int k = 0; k < this->x_gf_.at(compo).Size(); k++) {
246  xj[k] += this->x_gf_.at(compo)(k);
247  }
248  }
249 
250  mult_vectors(mobo, xj, mob);
251  if (this->single_phase_) {
252  for (int k = 0; k < mobi.Size(); k++) {
253  mobj[k] = mobi(k) * xo[k];
254  }
255  } else {
256  for (int k = 0; k < mobi.Size(); k++) {
257  mobj[k] = (mobi(k) * phi[k] + (1. - phi[k]) * this->secondary_mob_gf_.at(compo)(k)) * xo[k];
258  }
259  }
260  sum_vectors(mob, mobj, mob);
261  mult_vectors(mob, xj, mob);
262  mult_vectors(mob, xo, mob);
263  sum_vectors(sum_mob, mob, sum_mob);
264  scale_vector(mob, -1.0);
265  vmob.emplace_back(mob);
266  }
267  vmob.insert(vmob.begin(), sum_mob);
269  int j = 0;
270  for (auto& [output_infos, output_value] : output_system) {
271  mfem::Vector vv(vmob[j].size());
272  for (unsigned int k = 0; k < vmob[j].size(); k++) {
273  vv(k) = vmob[j][k];
274  std::string msg = "Error for j " + std::to_string(j) + " at node " + std::to_string(k) +
275  " with mob " + std::to_string(vv(k)) + " phi " + std::to_string(phi[k]);
276  MFEM_VERIFY(vv(k) < 1., msg.c_str());
277  }
278  output_value.get() = vv;
279  j++;
280  }
