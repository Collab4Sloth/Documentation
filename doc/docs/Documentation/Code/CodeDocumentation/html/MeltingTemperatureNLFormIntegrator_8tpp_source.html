70 /**
71  * @brief Check variables consistency
72  *
73  * @tparam VARS Template parameter defining the variables used
74  * in the integrator.
75  */
76 template <class VARS>
77 void MeltingTemperatureNLFormIntegrator<VARS>::check_variables_consistency() {
78  // Temperature scaling for mobility
79  this->temp_index_ = -1;
80  for (std::size_t i = 0; i < this->aux_infos_.size(); ++i) {
81  const auto& variable_info = this->aux_infos_[i];
82  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
83  size_t vsize = variable_info.size();
84 
85  MFEM_VERIFY(vsize >= 1,
86  "MeltingTemperatureNLFormIntegrator<VARS>: at least "
87  "one additionnal information is expected for auxiliary variables associated with "
88  "this integrator");
89  const std::string& symbol = toUpperCase(variable_info.back());
90  if (symbol == "T") {
91  this->temp_index_ = i;
92  break;
93  }
94  }
95 }
